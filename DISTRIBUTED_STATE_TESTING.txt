## Integration Example

```go
package handlers

import (
    "context"
    "example.com/axiomnizam/internal/distributedstate"
)

func InitializeDistributedState() (*distributedstate.DistributedManager, error) {
    store, err := distributedstate.NewEtcdStateStore([]string{"localhost:2379"})
    if err != nil {
        return nil, err
    }
    
    manager := distributedstate.NewDistributedManager(store, "axiom")
    return manager, nil
}

// Use in your handlers
manager, _ := InitializeDistributedState()
manager.PutState(ctx, "workflow/state", "running")
```
